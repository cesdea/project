CREATE TABLE QnA
   (	"B_ID" NUMBER(5,0), 
	"B_NAME" VARCHAR2(20 BYTE), 
	"B_EMAIL" VARCHAR2(50 BYTE), 
	"B_TITLE" VARCHAR2(80 BYTE), 
	"B_CONTENT" VARCHAR2(3000 BYTE), 
	"B_PWD" VARCHAR2(12 BYTE),
              "B_DATE" DATE, 
	"B_HIT" NUMBER(5,0), 
	"B_IP" VARCHAR2(20 BYTE), 
	"B_REF" NUMBER(5,0), 
	"B_STEP" NUMBER(5,0), 
	"B_LEVEL" NUMBER(5,0), 
	 PRIMARY KEY ("B_ID"));
